    Update entrypoints/content/index.tsx to integrate all components:

    ```typescript
    import { createShadowRootUi } from 'wxt/utils/content-script-ui/shadow-root';
      import ReactDOM from 'react-dom/client';
      import { BagIcon } from './BagIcon';
      import { TabGrid } from './TabGrid';
      import { Toast } from './Toast';
      import type { SavedTab } from './types';

      interface ToastState {
        visible: boolean;
        message: string;
        type: 'success' | 'warning';
      }

      export default defineContentScript({
        matches: ['<all_urls>'],
        main() {
          const ui = createShadowRootUi({
            name: 'magic-bag',
            position: 'inline',
            onMount: (container, shadow, shadowHost) => {
              // Inject styles
              const style = document.createElement('style');
              style.textContent = `
                @import url('/entrypoints/content/style.css');
              `;
              shadow.appendChild(style);

              // Create React root
              const root = ReactDOM.createRoot(container);
              let isGridOpen = false;
              let toastState: ToastState = { visible: false, message: '', type: 'success' };

              // Handle tab click - open URL in new tab (GRID-04)
              const handleTabClick = (url: string) => {
                chrome.tabs.create({ url });
              };

              // Toggle grid open/closed (GRID-01)
              const toggleGrid = () => {
                isGridOpen = !isGridOpen;
                render();
              };

              // Listen for toast messages from background script
              chrome.runtime.onMessage.addListener((message) => {
                if (message.type === 'success-toast') {
                  toastState = { visible: true, message: '已收入法宝袋', type: 'success' };
                  render();
                  setTimeout(() => {
                    toastState = { ...toastState, visible: false };
                    render();
                  }, 2000);
                } else if (message.type === 'duplicate-warning') {
                  toastState = { visible: true, message: '该标签页已在法宝袋中', type: 'warning' };
                  render();
                  setTimeout(() => {
                    toastState = { ...toastState, visible: false };
                    render();
                  }, 2000);
                }
              });

              // Render function
              const render = () => {
                root.render(
                  <>
                    <BagIcon onToggleGrid={toggleGrid} />
                    <TabGrid
                      isOpen={isGridOpen}
                      onClose={() => {
                        isGridOpen = false;
                        render();
                      }}
                      onTabClick={handleTabClick}
                    />
                    {toastState.visible && (
                      <Toast
                        message={toastState.message}
                        type={toastState.type}
                        onClose={() => {
                          toastState = { ...toastState, visible: false };
                          render();
                        }}
                      />
                    )}
                  </>
                );
              };

              // Initial render
              render();
            },
            onRemove: (container, shadow, shadowHost) => {
              // Clean up React root
              const root = shadow.getRootNode() as any;
              if (root && root.unmount) {
                root.unmount();
              }
            },
          });

          ui.mount();
        },
      });
    ```

    Key integrations:
    - BagIcon receives toggleGrid callback to open/close grid
    - TabGrid receives isOpen, onClose, onTabClick props
    - Toast renders conditionally based on toastState.visible
    - chrome.runtime.onMessage listener for background script messages
    - chrome.tabs.create for tab restoration (GRID-04)
    - All components render in Shadow DOM via createShadowRootUi
  </action>
