**Export implementation (per D-01):**
```typescript
const handleExport = async () => {
  const tabs = await savedTabs.getValue();
  const json = JSON.stringify(tabs, null, 2);
  const blob = new Blob([json], { type: 'application/json' });
  const url = URL.createObjectURL(blob);

  // Use chrome.downloads.download API
  chrome.downloads.download({
    url,
    filename: `magic-bag-tabs-${new Date().toISOString().split('T')[0]}.json`,
    saveAs: true
  });

  setToast({ visible: true, message: `已导出 ${tabs.length} 个标签页`, type: 'success' });
};
```

**Import implementation (per D-02, D-03):**
```typescript
const handleImport = async (event: React.ChangeEvent<HTMLInputElement>) => {
  const file = event.target.files?.[0];
  if (!file) return;

  try {
    const text = await file.text();
    const importedTabs: SavedTab[] = JSON.parse(text);

    let importedCount = 0;
    for (const tab of importedTabs) {
      const added = await addTab(tab); // addTab handles duplicate check
      if (added) importedCount++;
    }

    setToast({ visible: true, message: `已导入 ${importedCount} 个标签页`, type: 'success' });
  } catch (error) {
    setToast({ visible: true, message: `导入失败：${error}`, type: 'warning' });
  }
};
```
