**Purpose:** Enable users to collect tabs into the bag via familiar browser context menu. This is the primary interaction pattern for saving tabs.

**Output:** Context menu registered in background script, tab collection handler with duplicate detection, toast notification integration, and test coverage.

```typescript
// entrypoints/background.ts
export default defineBackground(() => {
  chrome.runtime.onInstalled.addListener(() => {
    chrome.contextMenus.create({
      id: 'save-tab-to-bag',
      title: '将标签页收入法宝袋',
      contexts: ['page'],
    });
  });

  chrome.contextMenus.onClicked.addListener(async (info, tab) => {
    if (info.menuItemId === 'save-tab-to-bag' && tab?.url) {
      await collectTab(tab);
    }
  });
});
```

<!-- From UI-SPEC.md Interaction Patterns: Context Menu Collection -->
Flow:
1. User right-clicks anywhere on page
2. Chrome context menu appears with "将标签页收入法宝袋" option
3. User selects the option
4. Service worker captures current tab URL, title, favicon
5. Service worker checks for duplicate URL
6. If duplicate: Show warning toast, do not add
7. If new: Add to storage (newest-first order), close original tab, show success toast

    Add to entrypoints/background.ts:

    ```typescript
    import { addTab } from '~/utils/storage';
    import type { SavedTab } from '~/entrypoints/content/types';

    export default defineBackground(() => {
      console.log('Background script started');

      chrome.runtime.onInstalled.addListener(() => {
        chrome.contextMenus.create({
          id: 'save-tab-to-bag',
          title: '将标签页收入法宝袋',
          contexts: ['page'],
        });
      });

      // Handle context menu clicks (per RESEARCH.md Pattern 1)
      chrome.contextMenus.onClicked.addListener(async (info, tab) => {
        if (info.menuItemId === 'save-tab-to-bag' && tab?.url) {
          await collectTab(tab);
        }
      });
    });

    async function collectTab(tab: chrome.tabs.Tab) {
      if (!tab.url || !tab.id) return;

      // Capture tab data (per CONTEXT.md D-08)
      const tabData: SavedTab = {
        url: tab.url,
        title: tab.title || 'Untitled',
        favicon: tab.favIconUrl,
        timestamp: Date.now(),
      };

      // Save to storage (addTab handles duplicate check per D-09, newest-first per D-10)
      const added = await addTab(tabData);

      if (added) {
        // Close original tab (per COLL-03)
        await chrome.tabs.remove(tab.id);

        // Send success toast to content script with error handling
        // Race condition: Tab might close before message arrives
        try {
          await chrome.tabs.sendMessage(tab.id, { type: 'success-toast' });
        } catch (error) {
          // Tab already closed or content script not ready - silent failure
          // This is expected behavior when tab closes immediately
          console.debug('Toast message not sent (tab closed):', error);
        }
      } else {
        // Send duplicate warning to content script with error handling
        try {
          await chrome.tabs.sendMessage(tab.id, { type: 'duplicate-warning' });
        } catch (error) {
          // Content script not ready - silent failure
          console.debug('Warning toast not sent:', error);
        }
      }
    }
    ```
